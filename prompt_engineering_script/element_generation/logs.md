using UnityEngine;

public class MyAnimation : MonoBehaviour
{
    private Transform root;
    private Transform pelvis, leftHip, rightHip, leftKnee, rightKnee, leftAnkle, rightAnkle;
    private Transform leftShoulder, rightShoulder, leftElbow, rightElbow, leftWrist, rightWrist;
    private float walkSpeed = 1.5f;
    private bool isWalking = false;
    private float walkCycleTime = 1f;  // Time for a full walk cycle (two steps)

    void Start()
    {
        // Automatically assigning the root joint
        root = transform.Find("m_avg_root");
        pelvis = root.Find("m_avg_Pelvis");
        leftHip = pelvis.Find("m_avg_L_Hip");
        rightHip = pelvis.Find("m_avg_R_Hip");
        leftKnee = leftHip.Find("m_avg_L_Knee");
        rightKnee = rightHip.Find("m_avg_R_Knee");
        leftAnkle = leftKnee.Find("m_avg_L_Ankle");
        rightAnkle = rightKnee.Find("m_avg_R_Ankle");
        leftShoulder = pelvis.Find("m_avg_Spine1/m_avg_Spine2/m_avg_Spine3/m_avg_L_Collar/m_avg_L_Shoulder");
        rightShoulder = pelvis.Find("m_avg_Spine1/m_avg_Spine2/m_avg_Spine3/m_avg_R_Collar/m_avg_R_Shoulder");
        leftElbow = leftShoulder.Find("m_avg_L_Elbow");
        rightElbow = rightShoulder.Find("m_avg_R_Elbow");
        leftWrist = leftElbow.Find("m_avg_L_Wrist");
        rightWrist = rightElbow.Find("m_avg_R_Wrist");
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.Space))
        {
            isWalking = !isWalking;
        }

        if (isWalking)
        {
            float time = Time.time % walkCycleTime;
            float walkPhase = time / walkCycleTime;

            WalkMovement(walkPhase);
        }
    }

    void WalkMovement(float phase)
    {
        // Walking consists of two half-cycles: one for each leg.
        // The legs and arms should move in opposite direction.

        // Calculate positions and rotations based on the walk cycle phase
        float strideAngle = Mathf.Sin(phase * 2 * Mathf.PI) * 20.0f;
        float kneeAngle = Mathf.Max(0, Mathf.Sin((phase + 0.5f) * 2 * Mathf.PI)) * 45.0f;
        float armSwingAngle = Mathf.Sin(phase * 2 * Mathf.PI) * 30.0f;

        // Move hips forward/backward
        leftHip.localRotation = Quaternion.Euler(-strideAngle, 0, 0);
        rightHip.localRotation = Quaternion.Euler(strideAngle, 0, 0);

        // Bend knees
        leftKnee.localRotation = Quaternion.Euler(kneeAngle, 0, 0);
        rightKnee.localRotation = Quaternion.Euler(kneeAngle, 0, 0);

        // Rotate ankles to point feet down/up
        leftAnkle.localRotation = Quaternion.Euler(-kneeAngle / 2, 0, 0);
        rightAnkle.localRotation = Quaternion.Euler(-kneeAngle / 2, 0, 0);

        // Swing arms opposite to legs
        leftShoulder.localRotation = Quaternion.Euler(0, 0, -armSwingAngle);
        rightShoulder.localRotation = Quaternion.Euler(0, 0, armSwingAngle);

        // Move the whole body forward
        root.position += root.forward * walkSpeed * Time.deltaTime;

        // Ensure the root's Y position never goes below 0
        if (root.position.y < 0)
        {
            root.position = new Vector3(root.position.x, 0, root.position.z);
        }

        // Optionally, you can add slight up and down motion if desired
        float bobAmount = Mathf.Abs(Mathf.Sin(phase * 2 * Mathf.PI)) * 0.05f;
        root.position = new Vector3(root.position.x, 0.05f + bobAmount, root.position.z);
    }
}
